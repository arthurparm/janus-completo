<MANDATORY_RULES>
1. **ONLY use labels and relationships defined in the schema above**
2. **ALWAYS use LIMIT** to prevent large result sets (default: 10, max: 50)
3. **ALWAYS return relevant properties**, not just node IDs
4. **Use OPTIONAL MATCH** for relationships that might not exist
5. **Use COALESCE** for properties that might be null
6. **Avoid Cartesian products** - always connect patterns
7. **Use proper variable naming** (n, m for nodes; r for relationships)
</MANDATORY_RULES>

<QUERY_PATTERNS>

--- Pattern 1: Find entity by name ---
MATCH (n:Concept {name: $name}) RETURN n.name, n.description LIMIT 1

--- Pattern 2: Find relationships of an entity ---
MATCH (n:Technology {name: $name})-[r]->(m)
RETURN n.name, type(r) AS relation, labels(m) AS target_type, m.name
LIMIT 20

--- Pattern 3: Find path between entities ---
MATCH path = shortestPath((a:Concept {name: $from})-[*..5]-(b:Concept {name: $to}))
RETURN [node in nodes(path) | node.name] AS path_nodes,
       [rel in relationships(path) | type(rel)] AS relations

--- Pattern 4: Find solutions to an error ---
MATCH (e:Error)-[:SOLVED_BY]->(s:Solution)
WHERE e.name CONTAINS $error_pattern OR e.message CONTAINS $error_pattern
RETURN e.name AS error, s.name AS solution, s.description AS how_to_fix
LIMIT 5

--- Pattern 5: Explore concepts related to a technology ---
MATCH (t:Technology {name: $tech_name})-[:USES|DEPENDS_ON|ENABLES*1..2]-(related)
RETURN DISTINCT labels(related)[0] AS type, related.name AS name, 
       related.description AS description
LIMIT 15

--- Pattern 6: Find function calls in a file ---
MATCH (f:File {path: $file_path})-[:CONTAINS]->(fn:Function)
OPTIONAL MATCH (fn)-[:CALLS]->(called:Function)
RETURN fn.name AS function, 
       COLLECT(DISTINCT called.name) AS calls_to
LIMIT 20

--- Pattern 7: User goals and progress ---
MATCH (u:User {name: $user})-[:HAS_GOAL]->(g:Goal)
OPTIONAL MATCH (g)-[:IMPLEMENTED_BY]->(p:Plan)
OPTIONAL MATCH (g)-[:BLOCKED_BY]->(i:Issue)
RETURN g.description AS goal, g.status AS status,
       p.id AS plan_id, i.description AS blocker
LIMIT 10

--- Pattern 8: Recent episodes/experiences ---
MATCH (e:Episode)
WHERE e.timestamp > datetime() - duration({hours: $hours})
RETURN e.id, e.summary, e.timestamp
ORDER BY e.timestamp DESC
LIMIT 20

--- Pattern 9: Technologies used by a concept ---
MATCH (c:Concept)-[:USES]->(t:Technology)
WHERE c.name CONTAINS $concept_name
RETURN c.name AS concept, COLLECT(t.name) AS technologies
LIMIT 10

--- Pattern 10: Error causation chain ---
MATCH (root:Error)-[:CAUSES*1..3]->(effect:Error)
WHERE root.name CONTAINS $error_name
RETURN root.name AS root_cause, 
       [n IN nodes(path) | n.name] AS error_chain
LIMIT 5
</QUERY_PATTERNS>

<ANTI_PATTERNS_TO_AVOID>
❌ MATCH (n) RETURN n                          // No filter = returns everything
❌ MATCH (n)-[*]-(m)                           // Unbounded traversal = explosion
❌ MATCH (n), (m) WHERE ...                    // Cartesian product
❌ MATCH (n:NonExistentLabel)                  // Label not in schema
❌ MATCH (n)-[:FAKE_RELATIONSHIP]->(m)         // Relationship not in schema
❌ RETURN *                                     // Too much data
❌ No LIMIT clause                              // Dangerous for large graphs
</ANTI_PATTERNS_TO_AVOID>
